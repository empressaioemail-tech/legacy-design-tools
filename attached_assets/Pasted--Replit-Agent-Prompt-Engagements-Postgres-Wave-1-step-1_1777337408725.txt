import "dotenv/config";
import { Pool } from "pg";
import { drizzle } from "drizzle-orm/node-postgres";
import * as schema from "./schema";

if (!process.env.DATABASE_URL) {
  throw new Error("DATABASE_URL not set. Add it in Replit Secrets.");
}

// TCP pool — explicitly NOT @neondatabase/serverless / WebSocket driver
export const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
  max: 5,
  idleTimeoutMillis: 30000,
  connectionTimeoutMillis: 5000,
});

export const db = drizzle(pool, { schema });

// Health probe for the API
export async function dbHealthCheck(): Promise<boolean> {
  try {
    const client = await pool.connect();
    try {
      await client.query("SELECT 1");
      return true;
    } finally {
      client.release();
    }
  } catch {
    return false;
  }
}

SECTION 4 — SCHEMA (apps/design-tools-web/server/db/schema.ts)
============================================================

import { pgTable, uuid, text, timestamp, jsonb, integer, index } from "drizzle-orm/pg-core";
import { relations } from "drizzle-orm";

export const engagements = pgTable("engagements", {
  id:           uuid("id").defaultRandom().primaryKey(),
  name:         text("name").notNull(),                  // matches Revit projectName
  nameLower:    text("name_lower").notNull().unique(),   // for case-insensitive lookup
  jurisdiction: text("jurisdiction"),                    // e.g., "Moab, UT" — nullable
  address:      text("address"),                         // nullable
  status:       text("status").notNull().default("active"), // active | on_hold | archived
  createdAt:    timestamp("created_at", { withTimezone: true }).defaultNow().notNull(),
  updatedAt:    timestamp("updated_at", { withTimezone: true }).defaultNow().notNull(),
}, (t) => ({
  nameLowerIdx: index("engagements_name_lower_idx").on(t.nameLower),
}));

export const snapshots = pgTable("snapshots", {
  id:            uuid("id").defaultRandom().primaryKey(),
  engagementId:  uuid("engagement_id").notNull().references(() => engagements.id, { onDelete: "cascade" }),
  projectName:   text("project_name").notNull(),
  payload:       jsonb("payload").notNull(),     // full snapshot JSON from Revit
  sheetCount:    integer("sheet_count"),
  roomCount:     integer("room_count"),
  levelCount:    integer("level_count"),
  wallCount:     integer("wall_count"),
  receivedAt:    timestamp("received_at", { withTimezone: true }).defaultNow().notNull(),
}, (t) => ({
  engagementIdx: index("snapshots_engagement_idx").on(t.engagementId),
  receivedAtIdx: index("snapshots_received_at_idx").on(t.receivedAt),
}));

export const engagementsRelations = relations(engagements, ({ many }) => ({
  snapshots: many(snapshots),
}));

export const snapshotsRelations = relations(snapshots, ({ one }) => ({
  engagement: one(engagements, {
    fields: [snapshots.engagementId],
    references: [engagements.id],
  }),
}));

export type Engagement = typeof engagements.$inferSelect;
export type NewEngagement = typeof engagements.$inferInsert;
export type Snapshot = typeof snapshots.$inferSelect;
export type NewSnapshot = typeof snapshots.$inferInsert;

import "dotenv/config";
import { db, pool } from "./index";
import { engagements } from "./schema";
import { sql } from "drizzle-orm";

async function main() {
  console.log("Seeding engagements…");

  // Idempotent: only insert if not already present
  const seedData = [
    {
      name: "Seguin Residence",
      nameLower: "seguin residence",
      jurisdiction: "Moab, UT",
      address: "1421 Seguin St, Moab, UT 84532",
      status: "active",
    },
    {
      name: "Musgrave Residence",
      nameLower: "musgrave residence",
      jurisdiction: "Moab, UT",
      address: "287 Musgrave Ln, Moab, UT 84532",
      status: "active",
    },
  ];

  for (const e of seedData) {
    await db.insert(engagements).values(e).onConflictDoNothing({ target: engagements.nameLower });
    console.log(`  ✓ ${e.name}`);
  }

  console.log("Done.");
  await pool.end();
}

main().catch((err) => {
  console.error(err);
  process.exit(1);

SECTION 6 — UPDATE EXPRESS SERVER (apps/design-tools-web/server/index.ts)
============================================================

Replace the in-memory Map entirely. Use Drizzle queries.

Endpoints:

1. GET /api/health
   - Returns { ok: true, db: "up" | "down" } based on dbHealthCheck()
   - Returns 503 if db is down

2. GET /api/engagements
   - Returns all engagements (id, name, jurisdiction, address, status, createdAt, updatedAt) ordered by updatedAt desc
   - For each, also include `latestSnapshot: { id, receivedAt, sheetCount, roomCount, levelCount, wallCount } | null` — the most recent snapshot
   - And `snapshotCount: number`
   - Use Drizzle subqueries or N+1 is fine for prototype (we have 2-20 engagements)

3. GET /api/engagements/:id
   - Returns the engagement plus its full snapshot list (newest first), each with id, projectName, sheetCount, roomCount, levelCount, wallCount, receivedAt — but NOT the full payload (keep response light)
   - 404 if not found

4. POST /api/snapshots
   - Auth: header `x-snapshot-secret` must equal env SNAPSHOT_SECRET, else 401
   - Body: arbitrary JSON; require projectName: string
   - Logic:
     a. Compute nameLower = projectName.trim().toLowerCase()
     b. Look up engagement by nameLower
     c. If not found, INSERT a new engagement with name = projectName, nameLower, jurisdiction = null, address = null, status = "active". Return the newly created engagement.
     d. Insert the snapshot:
        - engagementId = engagement.id
        - projectName from body
        - payload = the entire body
        - sheetCount = body.sheets?.length ?? null
        - roomCount = body.rooms?.length ?? null
        - levelCount = body.levels?.length ?? null
        - wallCount = body.walls?.count ?? body.walls?.length ?? null   (handle both old and new shapes)
     e. Update engagement.updatedAt = now()
   - Return: { id, receivedAt, engagementId, engagementName, autoCreated: boolean }

5. GET /api/snapshots/:id
   - Returns full snapshot row including payload, or 404

6. POST /api/chat
   - Body: { engagementId: string, question: string, history?: Array<{role, content}> }
   - Look up the engagement and its LATEST snapshot
   - If no snapshots exist for that engagement, return 400 with "No snapshots yet for this engagement. Send one from Revit first."
   - Build system prompt:
     "You are helping an architect understand their Revit model for the engagement '{engagement.name}'{ if engagement.address: ' at ' + engagement.address }{ if engagement.jurisdiction: ' (' + engagement.jurisdiction + ')' }. The most recent snapshot was captured {relative time, e.g. 'about 5 minutes ago'}.\n\nAnswer grounded in the snapshot data below. If the data does not contain what's asked, say so plainly. Be terse and operational in tone — this is a professional tool, not a chatbot.\n\n<snapshot received_at='{ISO}'>\n{JSON.stringify(payload, null, 2)}\n</snapshot>"
   - Stream the Claude response as SSE (model claude-sonnet-4-6, max_tokens 1024) — same pattern as before

Wrap startup in dbHealthCheck. If db is down at boot, log a clear error but still start the server (so /api/health can report it). Endpoints that need the DB return 503 if dbHealthCheck fails.

interface Engagement {
  id: string;
  name: string;
  jurisdiction: string | null;
  address: string | null;
  status: string;
  createdAt: string;
  updatedAt: string;
  latestSnapshot: {
    id: string;
    receivedAt: string;
    sheetCount: number | null;
    roomCount: number | null;
    levelCount: number | null;
    wallCount: number | null;
  } | null;
  snapshotCount: number;
}

interface SnapshotSummary { id: string; projectName: string; sheetCount: number | null; roomCount: number | null; levelCount: number | null; wallCount: number | null; receivedAt: string; }
interface SnapshotDetail extends SnapshotSummary { payload: any; }
interface ChatMessage { role: "user" | "assistant"; content: string; }

State:
- engagements: Engagement[]
- engagementsById: Record<string, { snapshots: SnapshotSummary[]; detail: Engagement }>
- snapshotDetailById: Record<string, SnapshotDetail>
- selectedSnapshotIdByEngagement: Record<string, string | null>  // which snapshot to view in detail panel; defaults to most recent
- messagesByEngagement: Record<string, ChatMessage[]>             // session-only, in-memory
- streaming: boolean

Actions:
- refreshEngagements(): GET /api/engagements
- loadEngagement(id): GET /api/engagements/:id
- loadSnapshotDetail(id): GET /api/snapshots/:id (cache in snapshotDetailById)
- selectSnapshot(engagementId, snapshotId)
- sendMessage(engagementId, question): POST /api/chat with SSE; appends user msg + streams assistant tokens

Auto-poll refreshEngagements every 5 seconds while EngagementList is mounted, and loadEngagement(id) every 5 seconds while EngagementDetail is mounted (so new snapshots from Revit appear live).
