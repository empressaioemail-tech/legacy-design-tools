{
  "name": "legacy-monorepo",

    "build": "npm run build --workspaces --if-present"

THEME BLOCK — packages/portal-ui/src/lib/theme.ts

export type ThemeName = "dark" | "light";

export function setTheme(name: ThemeName) {
  const html = document.documentElement;
  html.dataset.theme = name;
  html.classList.toggle("dark", name === "dark");
  localStorage.setItem("theme", name);
}

export function initTheme() {
  const saved = (localStorage.getItem("theme") as ThemeName) || "dark";
  setTheme(saved);
}

export function getTheme(): ThemeName {
  return (document.documentElement.dataset.theme as ThemeName) || "dark";
}

============================================================

Express server on port 3001. Endpoints:

1. POST /api/snapshots
   - Header `x-snapshot-secret` MUST equal env SNAPSHOT_SECRET, else 401
   - Body: arbitrary JSON; require at least { projectName: string }
   - Generate UUID, store in Map<string, { id, projectName, receivedAt: ISO string, payload: any }>
   - Return { id, receivedAt }

2. GET /api/snapshots
   - Returns array of { id, projectName, receivedAt } sorted newest first

3. GET /api/snapshots/:id
   - Returns full record or 404

4. POST /api/chat
   - Body: { snapshotId: string, question: string, history?: {role, content}[] }
   - Look up snapshot, build system prompt:
     "You are helping an architect understand their current Revit model. The model snapshot JSON is below. Answer grounded in this data; if the data doesn't contain what's asked, say so plainly. Be terse and operational in tone — this is a professional tool, not a chatbot.\n\n<snapshot>\n{JSON.stringify(snapshot, null, 2)}\n</snapshot>"
   - Call Anthropic API with model "claude-sonnet-4-6", max_tokens 1024, stream: true
   - Stream tokens back as SSE (text/event-stream); each event: `data: {"text": "..."}\n\n`; final event: `data: [DONE]\n\n`
   - Use env ANTHROPIC_API_KEY
